### 5.2 Lấy giờ hiện tại

```python
from datetime import datetime

def get_gio_hien_tai():
    """Lấy giờ âm lịch hiện tại"""
    now = datetime.now()
    hour = now.hour
    
    # Map giờ dương -> giờ âm
    if 23 <= hour < 1:
        return 'Tý'
    elif 1 <= hour < 3:
        return 'Sửu'
    elif 3 <= hour < 5:
        return 'Dần'
    elif 5 <= hour < 7:
        return 'Mão'
    elif 7 <= hour < 9:
        return 'Thìn'
    elif 9 <= hour < 11:
        return 'Tỵ'
    elif 11 <= hour < 13:
        return 'Ngọ'
    elif 13 <= hour < 15:
        return 'Mùi'
    elif 15 <= hour < 17:
        return 'Thân'
    elif 17 <= hour < 19:
        return 'Dậu'
    elif 19 <= hour < 21:
        return 'Tuất'
    else:
        return 'Hợi'
```
